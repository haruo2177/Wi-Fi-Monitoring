
```bash
# 実行権限があることを確認
ls -la mac.sh
# 必要に応じて権限設定
chmod +x mac.sh
```

```bash
# スクリプトの動作確認
./mac.sh

# ログファイルの出力確認
tail -1 network_monitor_log.csv

head -2 network_monitor_log.csv
```
